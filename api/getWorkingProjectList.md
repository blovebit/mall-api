# （平台/公司/设计师）获取在建工地列表: getWorkingProjectList

## 参数

    {
        "company_id": // 公司id，默认为空。
        "designer_id": // 设计师id，默认为空。
        "keyword": // 用户查询的关键字，默认为空。
        "process": // 默认全部 1:施工中；2：已竣工；
        "isrecommend": // 默认为空，获取全部；1为获取推荐的。
        "isindex": // 默认为空，获取全部。1为获取index的。
        "style": // 风格id。
        "unit": // 户型id。
        "page":
        "size":
    }

        "project_list": [

                    "id": int, // 设计师id

                },

                    "id": int, // 工长id
                    "name": "", // 工长名字
